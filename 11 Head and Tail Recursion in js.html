Head and Tail Recursion

function nums(x){
console.log(x);
if(x>0){
nums(x-1);
}
console.log(x);
}
let numData = 5;
nums(numData);

#1 // Here frist console before of this condition is called head recursion in this recursion condition will be solve one by one like: 5,4,3,2,1,0
